```prisma
model JournalEntry {
  id           String              @id @default(cuid())
  entityId     String
  entryNumber  String
  date         DateTime            @db.Date
  description  String
  status       JournalEntryStatus  @default(DRAFT)
  createdBy    String
  approvedBy   String?
  postedBy     String?
  createdAt    DateTime            @default(now())
  approvedAt   DateTime?
  postedAt     DateTime?
  reversalOfId String?             @unique
  lineItems    JournalEntryLine[]
  @@unique([entityId, entryNumber])
}

model JournalEntryLine {
  id             String   @id @default(cuid())
  journalEntryId String
  accountId      String
  debit          Decimal  @db.Decimal(19, 4)
  credit         Decimal  @db.Decimal(19, 4)
  memo           String?
  sortOrder      Int      @default(0)
}

enum JournalEntryStatus { DRAFT, APPROVED, POSTED }
```
