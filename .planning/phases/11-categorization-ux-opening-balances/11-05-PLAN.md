**Chosen approach:** In handleCategorize, when positionId is provided:
1. Fetch positions from `/api/entities/${entityId}/positions` (this will be cached by the browser / PositionPicker's module cache within 60s)

3. Construct positionLabel as `"${holdingName} -> ${name}"`
4. If not found (edge case), fall back to the account name

**Updated setCategorizePrompt block:**
```typescript
if (account && txn) {
  let resolvedPositionLabel: string | null = null;
  if (positionId) {
    try {
      const posRes = await fetch(`/api/entities/${entityId}/positions`);
      const posJson = await posRes.json();
      if (posJson.success) {
        const pos = posJson.data.find((p: { id: string }) => p.id === positionId);
        if (pos) {
          resolvedPositionLabel = `${pos.holdingName} -> ${pos.name}`;
        }
      }
    } catch { /* fallback to null */ }
  }

  setCategorizePrompt({
    transactionDescription: txn.description,
    merchantName: txn.merchantName,
    accountId,
    accountName: `${account.accountNumber} - ${account.name}`,
    positionId: positionId ?? null,
    positionLabel: resolvedPositionLabel,
  });
}
```
